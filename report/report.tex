  P\{GET\ EGGS\} &= \frac{\sum_{i}^{household}\sum_{j}^{day}GET\ EGGS_{ij}}{\sum_{i}^{household}\sum_{j}^{day}1} \\
  P\{GET\ EGGS |\ week\} &= 1 - P\{GET NO EGG\} \\
  &= 1 - (1 -  P\{GET\ EGGS\})^7 \\
  &= 0.126284
\end{align*}
